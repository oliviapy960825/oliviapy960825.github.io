#include "stdio.h"
int main(void) {
    int a, b, c;
    printf("Enter an integer:\n");
    scanf("%d", &a);
    printf("Enter an integer:\n");
    scanf("%d", &b);
    c = 0; 
    int i = 16;
    int bit_1 = 0;
    int bit0 = b & 0x0001;
    while (i > 0){
        if (bit_1 == 1 && bit0 == 0){
            c += a;
            
        }else if(bit_1 == 0 && bit0 == 1){
            c -= a;
        }
        a = a << 1;
        bit_1 = b & 0x001;
        b = b >> 1;
        bit0 = b & 1;
        i -= 1;
    }
    printf("the product = %d\n", c);
}
